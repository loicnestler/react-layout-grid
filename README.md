## Install

```bash
npm install --save react-layout-grid
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-layout-grid'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
